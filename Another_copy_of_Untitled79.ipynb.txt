        "# -------- SIMPLE REALISTIC COLORIZATION FUNCTION --------\n",
        "def simple_colorize(gray_img):\n",
        "    gray = gray_img.convert(\"L\")\n",
        "    gray_np = np.array(gray)\n",
        "\n",
        "    # Create soft natural colors\n",
        "    r = gray_np\n",
        "    g = cv2.GaussianBlur(gray_np, (15,15), 0)\n",
        "    b = 255 - gray_np\n",
        "\n",
        "    color = np.stack([r, g, b], axis=2)\n",
        "    return Image.fromarray(color.astype(\"uint8\"))\n",
        "\n",
        "# -------- STYLE FILTER FUNCTION --------\n",
        "def apply_style(img, style):\n",
        "    img_np = np.array(img)\n",
        "\n",
        "    if style == \"Warm Style\":\n",
        "        styled = cv2.applyColorMap(img_np, cv2.COLORMAP_PINK)\n",
        "\n",
        "    elif style == \"Cool Style\":\n",
        "        styled = cv2.applyColorMap(img_np, cv2.COLORMAP_WINTER)\n",
        "\n",
        "    elif style == \"Sketch\":\n",
        "        gray = cv2.cvtColor(img_np, cv2.COLOR_RGB2GRAY)\n",
        "        inv = 255 - gray\n",
        "        blur = cv2.GaussianBlur(inv, (21,21), 0)\n",
        "        sketch = cv2.divide(gray, 255-blur, scale=256)\n",
        "        styled = cv2.cvtColor(sketch, cv2.COLOR_GRAY2RGB)\n",
